-- 方式一
SELECT
    * 
FROM
    TABLE_TEST 
WHERE
    STATUS = '1' 
ORDER BY
    ORDER_NUMBER + 0 DESC
        
        
-- 方式二        
SELECT
    * 
FROM
    TABLE_TEST 
WHERE
    STATUS = '1' 
ORDER BY
    ORDER_NUMBER * 1 DESC
