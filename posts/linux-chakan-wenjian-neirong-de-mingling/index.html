head /etc/passwd    # 查看文件的前十行
head -2 /etc/passwd # 查看文件的前两行
tail /etc/passwd    # 查看文件的后十行
tail -f /etc/passwd # 实时跟踪文件的后十行
tail -2 /etc/passwd # 查看文件的后两行
wc -l /etc/passwd   # 显示文件的行数
nl /etc/passwd      # 直接打印文件内容并且显示行号
