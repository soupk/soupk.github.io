git init     # 初始化仓库
git status   # 查看工作区状态
git add -A   # 添加文件
git commit -m "提交信息"  # 将文件提交到repository里
git log      # 查看提交的记录
git diff     # 查询更改信息
git checkout -- .        # 撤销更改
git reset --hard 1df0573 # 回退，取版本号前7位就可以了
git clean -xf            # 删除新加的文件
