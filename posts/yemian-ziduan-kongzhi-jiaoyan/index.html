function checkIsNull() {
    var flag = false;
    var test1 = $('input:radio[name="test1"]:checked').val();
    var test2 = $("#test2").val()

    if (test1) {
        flag = true;
    } else {
        alert("输入框1不能为空！");
        flag = false;
    }
    if (flag && test2) {
        flag = true;
    } else if (flag) {
        alert("输入框2不能为空！");
        flag = false;
    }
    return flag;
}
