SELECT
 table_name,
 column_name
FROM
 information_schema. COLUMNS
WHERE
 table_schema = 'testdb'
AND table_name IN (
 SELECT
  TABLE_NAME
 FROM
  INFORMATION_SCHEMA. TABLES
 WHERE
  TABLE_SCHEMA = 'testdb'
)
ORDER BY
 table_name,
 column_name

SELECT
 A .table_name,
 A .COLUMN_NAME
FROM
 user_tab_columns A
WHERE
 TABLE_NAME IN (
  SELECT
   table_name
  FROM
   user_tables
 )
ORDER BY
 A .table_name

SELECT
 table_name,
 column_name
FROM
 information_schema.columns
WHERE
 table_name IN (
  SELECT
   TOP 100 PERCENT name
  FROM
   sysobjects
  WHERE
   xtype = 'u'
 )
ORDER BY
 table_name
